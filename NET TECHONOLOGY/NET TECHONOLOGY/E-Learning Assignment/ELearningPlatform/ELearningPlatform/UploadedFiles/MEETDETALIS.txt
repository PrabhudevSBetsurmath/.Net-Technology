create OR ALTER function USERDETAILS1(@ID INT)
RETURNS TABLE
AS
RETURN
select BookName,Edition,DateOfPublish,UserName 
from Books
Left JOIN Users 
ON Books.ID=Users.ID
where Users.id=@ID
go
